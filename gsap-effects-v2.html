Text Scramble
TEXT

Chữ bị "mã hóa" ngẫu nhiên rồi dần giải mã về đúng nội dung. Hiệu ứng hacker/decode cực ấn tượng cho headline.

<div id="scramble">TARGET</div>
#scramble{
  font-family:monospace;
  font-size:1.5rem;
  color:#23f5c8;
}
// Scramble effect (no plugin needed)
const chars = "!@#$%^&*ABCXYZ";
function scramble(el, target) {
  let iter = 0;
  const id = setInterval(() => {
    el.textContent = target.split("")
      .map((c, i) => i < iter
        ? target[i]
        : chars[Math.floor(Math.random()*chars.length)])
      .join("");
    if (iter >= target.length) clearInterval(id);
    iter += 0.4;
  }, 40);
}
scramble(document.getElementById("scramble"), "TARGET");

Particle Burst
IMAGE

Click tạo hàng loạt hạt bắn ra theo hướng ngẫu nhiên rồi fade out. Dùng GSAP stagger + random direction tạo hiệu ứng nổ pháo hoa.

<div class="burst-wrap">
  <div class="center" onclick="burst()"></div>
</div>
.particle{
  position:absolute;
  width:8px; height:8px;
  border-radius:50%;
  pointer-events:none;
}
function burst(e) {
  const colors = ["#7c6df5","#23f5c8","#f5a623","#f55296"];
  for(let i=0; i<20; i++){
    const dot = document.createElement("div");
    dot.className = "particle";
    dot.style.background = colors[i%colors.length];
    wrap.appendChild(dot);
    const a = (Math.random()*Math.PI*2);
    const d = gsap.utils.random(40,120);
    gsap.set(dot,{x:50%,y:50%});
    gsap.to(dot,{
      x:Math.cos(a)*d, y:Math.sin(a)*d,
      opacity:0, scale:0,
      duration:0.8, ease:"power2.out",
      onComplete:()=>dot.remove()
    });
  }
}

BUTTON

Sóng lan ra từ đúng điểm click. Tính vị trí chuột → tạo div → GSAP scale 0→1 + fade out. Material Design kinh điển nâng cấp.

<button class="ripple-btn"

  CLICK ME

.ripple-btn{
  position:relative;
  overflow:hidden;
}
.ripple-dot{
  position:absolute;
  border-radius:50%;
  background:rgba(124,109,245,0.4);
  pointer-events:none;
}
function ripple(e){
  const btn=e.currentTarget, r=btn.getBoundingClientRect();
  const dot=document.createElement("div");
  dot.className="ripple-dot";
  const size=Math.max(r.width,r.height)*2;
  gsap.set(dot,{width:size,height:size,
    x:e.clientX-r.left-size/2,
    y:e.clientY-r.top-size/2,scale:0});
  btn.appendChild(dot);
  gsap.to(dot,{scale:1,opacity:0,
    duration:0.6,ease:"power2.out",
    onComplete:()=>dot.remove()});
}

.elastic-btn{
  transform-origin:center;
  cursor:pointer;
}
function elasticClick(btn){
  gsap.timeline()
    .to(btn,{scaleX:0.85,scaleY:1.15,duration:0.1})
    .to(btn,{scaleX:1.1,scaleY:0.9,duration:0.1})
    .to(btn,{
      scale:1, duration:0.5,
      ease:"elastic.out(1, 0.4)"
    });
}

Overlay trượt vào (bottom→top) rồi ra (top→bottom) để chuyển trang. GSAP timeline với scaleY + transformOrigin — kỹ thuật agency chuẩn.

<div id="overlay"></div>
<a onclick="navigate('/new')">Link</a>
#overlay{
  position:fixed; inset:0;
  background:#7c6df5;
  transform:scaleY(0);
  transform-origin:bottom;
  z-index:9999;
}
function navigate(url){
  gsap.timeline()
    .to("#overlay",{scaleY:1,
      transformOrigin:"bottom",
      duration:0.6,ease:"power4.inOut"})
    .call(()=>window.location.href=url)
    .to("#overlay",{scaleY:0,
      transformOrigin:"top",
      duration:0.6,ease:"power4.inOut"});
}

Toast Notification
UI

Toast slide in từ phải rồi tự hide sau 2s. GSAP timeline: translateX + opacity in → hold → out. Chuẩn UX pattern cho feedback người dùng.

<div id="toast">✅ Thành công!</div>
<button onclick="showToast()">Notify</button>
#toast{
  position:fixed;
  top:20px; right:20px;
  transform:translateX(120%);
  opacity:0;
}
function showToast(){
  gsap.timeline()
    .to("#toast", {
      x:0, opacity:1,
      duration:0.4,
      ease:"back.out(1.5)"
    })
    .to("#toast", {
      x:"120%", opacity:0,
      duration:0.3,
      ease:"power2.in",
      delay:2
    });
}
